д»Јз Ғ

class Solution {
    int gcd(int a, int b) {
        if(b == 0) return a;
        return gcd(b, a%b);
    }
public:
    int countDifferentSubsequenceGCDs(vector<int>& nums) {
        int ma = *max_element(nums.begin(), nums.end());
        vector<bool> occur(ma+1, false);
        for(int i = 0; i < nums.size(); i++) {
            occur[nums[i]] = true;
        }
        int ans = 0;
        // д»Һ[1, ma]дёӯжһҡдёҫжңҖеӨ§е…¬зәҰж•°
        for(int i = 1; i <= ma; i++) {
            int subgcd = 0;
            for(int j = i; j <= ma; j += i) {
                if(!occur[j]) continue;
                if(subgcd == 0) subgcd = j;
                else subgcd = gcd(subgcd, j);
                if(subgcd == i) {
                    ans++;
                    break;
                }
            }
        }
        return ans;
    }
};
